\begin{frame}{Vector Quantization Strategies}
	\begin{columns}
		\column{0.48\textwidth}
		\begin{exampleblock}{Compression techniques}
			\begin{itemize}
				\item \textbf{Scalar Quantization}: float32 $\rightarrow$ uint8, giảm RAM khoảng 4x.
				\item \textbf{Product Quantization}: chia vector thành sub-vectors, dùng centroids, giảm 8--64x.
				\item \textbf{Binary Quantization}: vector $\rightarrow$ binary bits, tăng tốc comparison với trade-off accuracy.
			\end{itemize}

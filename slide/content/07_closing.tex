\begin{frame}{Developer Experience Score Matrix}
	\begin{table}
		\centering
		\caption{DX matrix cho nhóm xây dựng RAG application.}
		\resizebox{\textwidth}{!}{%
		\begin{tabular}{lccc}
			\hline
			\textbf{Tiêu chí DX} & \textbf{Qdrant} & \textbf{Weaviate} & \textbf{Milvus} \\
			\hline
			Local setup & rất dễ & dễ & khó hơn \\
			Dependency count & thấp & thấp-vừa & cao \\
			Query API ergonomics & rõ ràng & rất tốt cho RAG & mạnh nhưng nghiêm ngặt \\
			Schema evolution & linh hoạt payload & auto/schema-driven & strict FieldSchema \\
			Observability burden & thấp & trung bình & cao \\
			Production operations & nhẹ & vừa & nặng nhưng scale tốt \\
			\hline
		\end{tabular}}
	\end{table}
	\begin{block}{DX takeaway}
		DX bao gồm setup, mental model, debugging, schema evolution, dependency graph và monitoring, không chỉ là số dòng code.
	\end{block}
	\note{Developer Experience là yếu tố thực dụng khi chọn Vector Database. Qdrant dễ đưa vào prototype và service nhỏ. Weaviate dễ xây RAG app có Hybrid Search. Milvus cần nhiều kiến thức vận hành hơn nhưng phù hợp khi scale và HA là yêu cầu trung tâm.}
\end{frame}

\section{Đánh giá thực nghiệm}

\begin{frame}{Giao thức benchmark công bằng}
	\begin{columns}
		\column{0.52\textwidth}
		\begin{block}{Biến được kiểm soát}
			\begin{itemize}
				\item Cùng corpus, chunk size, overlap và preprocessing.
				\item Cùng embedding model và vector dimension.
				\item Cùng query set, Top-K và random seed.
				\item Cùng hardware envelope, Docker limits và warm-up policy.
			\end{itemize}
		\end{block}
		\column{0.48\textwidth}
		\begin{exampleblock}{Metric đo lường}
			Latency p50/p95, RAM footprint, insert/build cost, Recall@K, MRR và DX score.
		\end{exampleblock}
	\end{columns}
	\vspace{0.08cm}
	\scriptsize\textbf{Hardware:} AMD Ryzen 7 5800H, 16 logical CPUs, 14GiB RAM, NVMe SSD 512GB, Arch Linux.\\
	\textbf{Nguyên tắc diễn giải:} mỗi database có lifecycle và planner riêng; benchmark phải ghi rõ config. Không tính bước cài đặt, đăng ký tài khoản hay thao tác cloud console.
	\note{Benchmark công bằng không có nghĩa là ép mọi hệ thống chạy giống hệt nhau, vì kiến trúc của chúng khác nhau. Điều quan trọng là cùng dữ liệu, cùng query và cùng mục tiêu retrieval, đồng thời ghi rõ lifecycle riêng như flush/load của Milvus hoặc Hybrid Search của Weaviate.}
\end{frame}

\begin{frame}{Độ chính xác truy hồi: Recall@K \& MRR}
	\tiny
	\begin{columns}
		\column{0.34\textwidth}
		\begin{exampleblock}{Định nghĩa metric}
			\[
				\mathrm{Recall@K}=\frac{|R_K|}{|R|}
			\]
			\[
				MRR=\frac{1}{|Q|}\sum_{q\in Q}\frac{1}{\operatorname{rank}_{q}}
			\]
		\end{exampleblock}
		\begin{alertblock}{Rủi ro}
			Latency thấp nhưng truy hồi sai làm LLM thiếu evidence đáng tin cậy.
		\end{alertblock}
		\vspace{0.08cm}
		\textbf{Nhận xét:} Milvus có Recall@K/MRR cao nhất trong tập thử nghiệm; Qdrant và Weaviate cần tune thêm Top-K, hybrid alpha hoặc index params.
		\column{0.66\textwidth}
			\begin{figure}
				\centering
				\includegraphics[width=\linewidth]{accuracy-chart}
				\caption{Recall@K và MRR từ dashboard benchmark.}
			\end{figure}
		\end{columns}
		\note{Với RAG, metric quan trọng không chỉ là database trả lời nhanh. Nếu Top-K chunks không chứa evidence đúng, LLM sẽ không có cơ sở để trả lời. Corpus benchmark ở đây là synthetic, nhưng embedding và database path chạy thật khi MOCK_MODE=false. Vì vậy kết quả phù hợp để so sánh cùng workload, không nên diễn giải thành chất lượng tuyệt đối trên mọi tài liệu production.}
	\end{frame}

\begin{frame}{Đánh đổi accuracy-speed \& Pareto Frontier}
	\scriptsize
	\begin{columns}
		\column{0.34\textwidth}
		\begin{block}{Cách đọc Pareto}
			\begin{itemize}
				\item Trục X: latency trung bình.
				\item Trục Y: Recall@K.
				\item Điểm tốt nằm cao và lệch trái.
				\item Chọn cấu hình theo SLA thay vì nhìn một metric đơn lẻ.
			\end{itemize}
		\end{block}
		\vspace{0.1cm}
		\textbf{Ý nghĩa:} Milvus thắng về recall trong snapshot; Qdrant/Weaviate phù hợp hơn khi ngân sách latency chặt.
		\column{0.66\textwidth}
			\begin{figure}
				\centering
				\includegraphics[width=\linewidth]{tradeoff-chart}
				\caption{Pareto Frontier thực nghiệm: accuracy so với latency trung bình.}
			\end{figure}
		\end{columns}
		\note{Pareto Frontier giúp tránh kết luận một chiều. Một cấu hình nhanh hơn nhưng Recall thấp hơn chưa chắc tốt. Một cấu hình chính xác hơn nhưng latency vượt SLA cũng không phù hợp. Biểu đồ này dùng AvgLatency_ms từ tradeoff.csv, không phải p95. Cần chọn điểm cân bằng theo use case, corpus size và yêu cầu production.}
	\end{frame}

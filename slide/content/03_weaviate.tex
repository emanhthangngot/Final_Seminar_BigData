\begin{frame}{Hybrid Search: kết hợp keyword và semantic retrieval}
	\begin{columns}
		\column{0.34\textwidth}
		\begin{block}{BM25F}
			Mạnh với keyword chính xác, tên riêng, mã lỗi, thuật ngữ hiếm.
		\end{block}
		\column{0.34\textwidth}
		\begin{exampleblock}{Vector}
			Mạnh với paraphrase, đồng nghĩa, truy vấn theo ý nghĩa.
		\end{exampleblock}
		\column{0.32\textwidth}
		\begin{alertblock}{$\alpha$}
			$0.0$ = keyword\\
			$0.5$ = cân bằng\\
			$1.0$ = vector
		\end{alertblock}
	\end{columns}
	\takeaway{Weaviate phù hợp khi tài liệu vừa cần exact terms, vừa cần semantic similarity.}
	\note{Hybrid Search giải quyết điểm yếu riêng của từng phương pháp. Vector search có thể bỏ sót mã định danh chính xác; BM25 có thể bỏ sót paraphrase. Weaviate cho phép kết hợp hai tín hiệu này trong một API call, rất phù hợp RAG trên tài liệu kỹ thuật.}
\end{frame}

\begin{frame}{RRF: hợp nhất hai bảng xếp hạng}
	\begin{exampleblock}{Reciprocal Rank Fusion}
		\[
			\operatorname{score}(d)=\sum_{i=1}^{m}\frac{1}{k+\operatorname{rank}_{i}(d)}
		\]
	\end{exampleblock}
	\begin{columns}
		\column{0.52\textwidth}
			\begin{table}
				\centering
				\caption{Ví dụ fusion từ BM25 và vector ranking.}
				\scriptsize
				\resizebox{\linewidth}{!}{%
				\begin{tabular}{lccc}
					\hline
					Document & BM25 rank & Vector rank & Ý nghĩa \\
				\hline
				$d_1$ & 1 & 5 & keyword rất mạnh \\
				$d_2$ & 8 & 1 & semantic rất mạnh \\
					$d_3$ & 20 & 18 & yếu ở cả hai \\
					\hline
				\end{tabular}}
			\end{table}
		\column{0.48\textwidth}
		\begin{block}{Rank position thay cho raw score}
			BM25 score và cosine similarity khác thang đo; RRF dùng vị trí xếp hạng nên ổn định hơn khi hợp nhất hai nguồn tín hiệu.
		\end{block}
	\end{columns}
	\note{Công thức RRF cộng đóng góp từ nhiều ranked lists. Document đứng càng cao ở một nguồn thì đóng góp càng lớn. Cách này tránh vấn đề phải chuẩn hóa điểm BM25 và vector similarity vốn có bản chất rất khác nhau.}
\end{frame}

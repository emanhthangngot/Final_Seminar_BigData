\begin{frame}{Vì sao cần Vector Database?}
	\begin{columns}[T]
		\column{0.33\textwidth}
		\begin{block}{LLM thiếu dữ liệu mới}
			Không tự biết tài liệu nội bộ, chính sách mới hoặc dữ liệu sau thời điểm huấn luyện.
		\end{block}
		\column{0.33\textwidth}
		\begin{block}{Cần evidence}
			RAG phải lấy đúng đoạn tài liệu liên quan trước khi sinh câu trả lời.
		\end{block}
		\column{0.33\textwidth}
		\begin{block}{Vector DB là tầng retrieval}
			Lưu embedding, tìm Top-K chunks gần nghĩa và đưa context vào prompt.
		\end{block}
	\end{columns}
	\takeaway{SQL/NoSQL mạnh ở exact match; Vector Database mạnh ở tìm kiếm theo ngữ nghĩa.}
	\note{Nhấn mạnh Vector Database không thay thế database truyền thống, mà giải bài toán khác: tìm theo ngữ nghĩa. Với RAG, chất lượng context phụ thuộc trực tiếp vào retrieval layer; nếu retrieval sai, LLM sẽ sinh câu trả lời sai dù model rất mạnh.}
\end{frame}

\begin{frame}{Kiến trúc pipeline RAG}
	\footnotesize
	\begin{columns}
		\column{0.36\textwidth}
		\begin{block}{Indexing}
			PDF $\rightarrow$ chunks $\rightarrow$ embeddings $\rightarrow$ Vector DB
		\end{block}
		\begin{exampleblock}{Answering}
			Question $\rightarrow$ Top-K context $\rightarrow$ LLM answer
		\end{exampleblock}
		\begin{alertblock}{Benchmark đo gì?}
			Latency, Recall@K, MRR, tradeoff và DX.
		\end{alertblock}
		\column{0.64\textwidth}
		\begin{figure}
			\centering
			\begin{tikzpicture}[node distance=0.34cm and 0.28cm, scale=0.84, transform shape]
				\node[flowbox, fill=softblue, draw=qdrantblue!65, minimum width=1.65cm] (pdf) {PDF};
				\node[flowbox, right=of pdf, minimum width=1.65cm] (chunks) {Chunks};
				\node[flowbox, right=of chunks, fill=softgreen, draw=weaviategreen!65, minimum width=1.9cm] (embed) {Embedding};
				\node[flowbox, right=of embed, fill=softorange, draw=milvusorange!65, minimum width=1.95cm] (db) {Vector DB};
				\node[flowbox, below=0.92cm of pdf, fill=softblue, draw=qdrantblue!65, minimum width=1.65cm] (q) {Question};
				\node[flowbox, right=of q, fill=softgreen, draw=weaviategreen!65, minimum width=1.75cm] (search) {ANN Search};
				\node[flowbox, right=of search, minimum width=1.75cm] (ctx) {Top-K};
				\node[flowbox, right=of ctx, fill=softorange, draw=milvusorange!65, minimum width=1.85cm] (llm) {LLM Answer};

				\draw[flowarrow] (chunks) -- (embed);

				\draw[flowarrow] (q) -- (search);
				\draw[flowarrow] (db.south) -- ++(0,-0.48) -| (search.north);
				\draw[flowarrow] (search) -- (ctx);
				\draw[flowarrow] (ctx) -- (llm);
			\end{tikzpicture}
			\caption{Index trước, retrieval khi người dùng hỏi.}
		\end{figure}
	\end{columns}
	\note{Giải thích hai pha của RAG. Ingestion biến dữ liệu thành embedding và metadata. Retrieval biến câu hỏi thành query vector, tìm nearest neighbors rồi đưa context vào prompt. Benchmark Vector Database phải đo cả tốc độ search lẫn độ đúng của chunks được lấy ra.}
\end{frame}

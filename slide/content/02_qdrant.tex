\section{Qdrant}

\begin{frame}{Qdrant: kiến trúc tối giản cho filtered retrieval}
	\begin{columns}
		\column{0.42\textwidth}
		\begin{block}{Đặc điểm triển khai}
			\begin{itemize}
				\item Single binary, ít dependency.
				\item Segment giữ vector, payload và HNSW.
				\item WAL bảo vệ write path.
			\end{itemize}
		\end{block}
		\begin{exampleblock}{Điểm cần nhớ}
			Qdrant tối ưu cho RAG có metadata filter như tenant, ACL, timestamp.
		\end{exampleblock}

\end{frame}

\begin{frame}{Qdrant và bài toán truy hồi có filter}
	\footnotesize
	\begin{columns}
		\column{0.52\textwidth}
		\begin{block}{Ý tưởng chính}
			\begin{itemize}
				\item HNSW là ANN index dạng đồ thị.
				\item Filter không chỉ được xử lý ở bước hậu xử lý.
				\item Qdrant đưa \textbf{payload filter} vào quá trình duyệt graph.
				\item Khi filter có độ chọn lọc cao, planner có thể đổi sang chiến lược phù hợp hơn.
			\end{itemize}
		\end{block}
			\begin{exampleblock}{Adaptive behavior}
					Filter phổ biến $\rightarrow$ Filterable HNSW.\\
					Filter rất hẹp $\rightarrow$ payload index + brute-force.
			\end{exampleblock}
		\column{0.48\textwidth}
		\begin{figure}
			\centering
				\begin{tikzpicture}[scale=0.86, transform shape, every node/.style={circle, draw=diagramgray!65, minimum size=5.5mm, font=\tiny}]

\end{frame}

\begin{frame}{Ba chiến lược kết hợp vector search và metadata}
	\begin{columns}[T]
		\column{0.33\textwidth}
		\begin{block}{Pre-filter}
			Lọc metadata trước rồi search trên tập con.\\[0.12cm]
			\textbf{Rủi ro:} graph bị thưa.
		\end{block}
		\column{0.33\textwidth}
		\begin{alertblock}{Post-filter}
			Search vector trước rồi loại kết quả sai filter.\\[0.12cm]
			\textbf{Rủi ro:} thiếu Top-K hợp lệ.
		\end{alertblock}
		\column{0.33\textwidth}
		\begin{exampleblock}{In-graph}
			Kiểm tra filter ngay khi duyệt HNSW.\\[0.12cm]
			\textbf{Hợp:} filter-heavy RAG.
		\end{exampleblock}
	\end{columns}
	\takeaway{Trong production RAG, filter ảnh hưởng trực tiếp đến Recall@K, latency và tenant isolation.}
	\note{Pre-filtering có thể làm graph bị thưa, post-filtering có thể trả về không đủ candidates hợp lệ. In-graph filtering giữ được lợi thế của graph traversal trong khi vẫn tôn trọng metadata constraints. Đây là lý do Qdrant thường phù hợp workload nhiều filter.}
\end{frame}
